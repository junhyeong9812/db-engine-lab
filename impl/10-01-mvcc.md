## 1. invariant (SI)
- TX는 자기 snapshot 시점의 committed 데이터만 봄.
- Repeatable read: 같은 read 두 번 → 같은 결과.
- 미커밋 tx 변경 invisible.

## 2. MVCCStore.kt — 한 줄 한 줄

```kotlin
package com.dbenginelab.mvcc                                         // 신규 mvcc 패키지

import java.util.Collections
import java.util.concurrent.ConcurrentHashMap
import java.util.concurrent.atomic.AtomicLong

class MVCCStore<K, V> {

    data class Version<V>(
        val value: V?,                                                // null = tombstone (DELETE)
        val xidStart: Long,                                           // 생성 tx
        @Volatile var xidEnd: Long = INFINITY,                        // 종료 tx (or INFINITY)
    ) {
        companion object { const val INFINITY: Long = Long.MAX_VALUE }
    }

    data class Snapshot(val xid: Long, val active: Set<Long>) {
        // Q: isVisible 5조건 각각이 무엇 보호?
        fun isVisible(v: Version<*>): Boolean {
            if (v.xidStart > xid) return false                       // (1) 미래 tx — invisible
            if (v.xidStart in active) return false                    // (2) 동시 미커밋 tx 변경 — invisible
            if (v.xidEnd == Version.INFINITY) return true             // (3) 아직 살아있음 — visible
            if (v.xidEnd > xid) return true                           // (4) 우리 시점 후 삭제 — 우리는 봄
            if (v.xidEnd in active) return true                       // (5) 삭제 tx 아직 active — 우리는 봄
            return false                                              // 우리 시점 전 commit된 삭제 — invisible
        }
        // <details><summary>A</summary>
        // SI 핵심 — "snapshot 시점에 committed인 가장 최근 visible version". active set 체크가 핵심.
        // </details>
    }

    private val chains: ConcurrentHashMap<K, MutableList<Version<V>>> = ConcurrentHashMap()

    @Synchronized
    fun insert(key: K, value: V, xid: Long) {
        val chain = chains.getOrPut(key) { mutableListOf() }
        // Q: 새 버전 추가 전 기존 살아있는 버전의 xidEnd 마킹?
        chain.firstOrNull { it.xidEnd == Version.INFINITY }?.xidEnd = xid
        chain.add(Version(value, xid))
        // <details><summary>A</summary>
        // Update semantics — 같은 key의 새 insert는 옛 버전을 terminate. 한 시점 한 visible version (SI invariant).
        // </details>
    }

    @Synchronized
    fun delete(key: K, xid: Long) {
        val chain = chains[key] ?: return
        chain.firstOrNull { it.xidEnd == Version.INFINITY }?.xidEnd = xid
        chain.add(Version(null, xid))                                 // tombstone
    }

    @Synchronized
    fun get(key: K, snapshot: Snapshot): V? {
        val chain = chains[key] ?: return null
        for (i in chain.indices.reversed()) {                         // 가장 최근부터 — 먼저 visible 만나면 OK
            val v = chain[i]
            if (snapshot.isVisible(v)) return v.value
        }
        return null
    }

    fun versionCount(key: K): Int = chains[key]?.size ?: 0
}

class SnapshotProvider {                                              // tx ID + active set 관리
    private val nextXid = AtomicLong(1)
    private val active: MutableSet<Long> = Collections.synchronizedSet(mutableSetOf())

    @Synchronized
    fun begin(): MVCCStore.Snapshot {
        val xid = nextXid.getAndIncrement()
        active.add(xid)
        return MVCCStore.Snapshot(xid, active.toSet() - xid)         // 자기 xid 제외 active
    }

    @Synchronized
    fun commit(snapshot: MVCCStore.Snapshot) {
        active.remove(snapshot.xid)
    }
}
```
